# [Unreal 1/2기] CH 2 간단한 프로그래밍  구현 1번 과제
## 도전 기능
- 입력 받은 숫자 배열을 정렬 하는 프로그램
- 오름차순 정렬과 내림차순 정렬을 구현
- 정렬은 각각 함수로 구현
- 숫자 1을 입력 받으면 오름차순 정렬, 숫자 2를 입력 받으면 내림차순 정렬 하도록 구현
- <b>algorithm 헤더의 sort 함수 사용x</b>

### 사용한 정렬 알고리즘
  > 다양한 알고리즘을 사용 해보려고 각 정렬 알고리즘을 달리 했음
- 오름차순 : 버블 정렬(Bubble Sort)
- 내림차순 : 삽입 정렬(Insertion Sort)

```c#
#include <iostream>
#include <vector>

using namespace std;

// 배열 출력 함수
void printArr(string printString, vector<int> arr)
{
	cout << printString << " 배열 : [";
	for (int i = 0; i < arr.size(); ++i)
	{
		cout << arr[i];
		if (i < arr.size() - 1)
		{
			cout << ", ";
		}
	}
	cout << "]" << endl;
}

// 버블 정렬 사용
// 오름차순
vector<int> sortUP(vector<int> arr)
{
	int key;

	// 2개의 수를 검사해서 큰 수가 뒤쪽으로 가고
	// 다음 루프 마다 자기 자신과 이전에 검사한 요소는 다시 검사할 필요가 없으므로
	// 점차 검사할 범위가 작아짐
	// 배열의 크기 = 5 일때,
	// 처음 검사 범위 : 4 => 한번 검사후 범위 => 3 ...
	for (int i = arr.size() - 1; i > 0; --i)
	{
		// 각 검사 범위 내에서 현재 요소와 다음 요소를 비교
		for (int j = 0; j < i; ++j)
		{
			// 현재 요소가 다음 요소보다 커지면 뒤로 이동
			if (arr[j] > arr[j + 1])
			{
				key = arr[j];
				arr[j] = arr[j + 1];
				arr[j + 1] = key;
			}
		}
	}

	return arr;
}

// 삽입 정렬 사용
// 내림차순
vector<int> sortDown(vector<int> arr)
{
	int key;

	// 인덱스 1부터 검사 (첫 번째 요소 제외)
	for (int i = 1; i < arr.size(); ++i) {
		
		// i번째 숫자를 key로 저장
		key = arr[i];

		// 첫 번째 부터 i - 1번째 까지 숫자는 검사 했으므로
		// 현재 i번째 값을 검사했던 i - 1(j)번째 ~ 첫 번째 사이 값들과 역순으로 비교
		// j가 0보다 크거나 같고, arr[j]가 key보다 작을 동안만 반복
		int j;
		for (j = i - 1; j >= 0 && arr[j] < key; --j) {
			// 반복되는 동안 한칸씩 뒤로 이동
			arr[j + 1] = arr[j];
		}

		arr[j + 1] = key;
	}

	return arr;
}

int main()
{
	// 입력 받을 숫자의 총 수;
	int n;

	cout << "입력할 숫자의 총 개수(양의 정수) 입력 : ";
	cin >> n;
	cout << endl;

	// 0과 음수 입력 방지
	if (n <= 0)
	{
		cout << "양의 정수를 입력하세요.";
		return 1;
	}

	// 입력 받은 숫자 배열
	vector<int> numbers;

	for (int i = 0; i < n; ++i)
	{
		// 입력 받은 숫자 저장할 임시 변수
		int number;
		cout << i + 1 << " 번째 수 입력 : ";
		cin >> number;

		numbers.push_back(number);
	}

	printArr("정렬 전", numbers);

	int sortOp = 1;
	cout << "정렬 동작 입력(오름차순 = 1, 내림차순 = 2) : ";
	cin >> sortOp;

	// 1과 2가 아닌 숫자 입력 방지
	if (sortOp < 1 || sortOp > 2)
	{
		cout << "1 또는 2를 입력하세요.";
		return 1;
	}

	// 입력 받은 숫자에 따라 오름차순 정렬, 내림차순 정렬 결정
	if (sortOp == 1)
	{
		numbers = sortUP(numbers);
	}
	else
	{
		numbers = sortDown(numbers);
	}

	printArr("정렬 후", numbers);

	return 0;
}
```
